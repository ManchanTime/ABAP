FORM get_header_data .

* Header 데이터 세팅
  CLEAR gs_header.
  SELECT SINGLE a~vbeln, a~kunnr, name1, bstnk, netwr,
                vdatu, b~zterm, b~inco1, waerk
    FROM vbak AS a LEFT JOIN vbkd AS b
                          ON a~vbeln = b~vbeln
                   LEFT JOIN vbkd AS c
                          ON a~vbeln = c~vbeln
                         AND c~posnr IS NULL
                   LEFT JOIN kna1 AS d
                          ON a~kunnr = d~kunnr
    WHERE a~vbeln = @p_vbeln
    INTO CORRESPONDING FIELDS OF @gs_header.

  IF sy-subrc <> 0.
    MESSAGE s101 DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

ENDFORM.

FORM export_excel .

  DATA : lv_path     TYPE string,
         lv_filename TYPE string.

* file 오브젝트 생성
  IF go_objfile IS INITIAL.
    TRY.
        CREATE OBJECT go_objfile.
      CATCH cx_root.
        STOP.
    ENDTRY.
  ENDIF.

* 기존 생성 폴더가 없다면 세팅
  IF gv_pfolder IS NOT INITIAL.
    gv_choosed_folder = gv_pfolder.
  ELSE.
    CALL METHOD cl_gui_frontend_services=>get_temp_directory
      CHANGING
        temp_dir             = gv_choosed_folder
      EXCEPTIONS
        cntl_error           = 1
        error_no_gui         = 2
        not_supported_by_gui = 3
        OTHERS               = 4.
  ENDIF.

* 파일 이름 세팅
  DATA(lv_file_name) = CONV string( TEXT-n01 ).
  REPLACE '&1' IN lv_file_name WITH gs_header-vbeln.
  REPLACE '&2' IN lv_file_name WITH sy-datum.

* 파일 선택 다이얼로그 출력
  CALL METHOD cl_gui_frontend_services=>file_save_dialog
    EXPORTING
*     default_extension         =
      default_file_name         = lv_file_name
      file_filter               = 'Excel files (*.xls;*.xlsx)|*.xlsx'
      initial_directory         = 'D:\'
    CHANGING
      filename                  = lv_filename
      path                      = lv_path
      fullpath                  = gv_choosed_folder
    EXCEPTIONS
      cntl_error                = 1
      error_no_gui              = 2
      not_supported_by_gui      = 3
      invalid_default_file_name = 4
      OTHERS                    = 5.

* 파일 위치 및 이름 세팅
  IF sy-subrc = 0.
    gv_pfolder = gv_choosed_folder.
  ELSE.
    MESSAGE s102 DISPLAY LIKE 'E'.
    RETURN.
  ENDIF.

* 파일 세팅 안하면 종료
  IF gv_pfolder IS INITIAL.
    RETURN.
  ENDIF.

* 파일 오브젝트 해제
  CALL METHOD : go_objfile->free.
  FREE : go_objfile.

  PERFORM download_excel.

ENDFORM.

FORM download_excel .

  DATA : lv_rc TYPE i.

*-- File name 세팅
  CLEAR gv_file.
  gv_file = gv_choosed_folder.

* 출력 Excel 파일
  gv_form = 'Z2508R0030_087'.

* 템플릿 세팅
  PERFORM download_template.
* 템플릿 워크북 및 시트 세팅
  PERFORM open_excel_template.

*---------------------------------------------------------------------
* 엑셀 데이터 세팅
*---------------------------------------------------------------------
* 회사정보 세팅
  PERFORM fill_company_data.
* SO 헤더 정보 세팅
  PERFORM fill_header_data.
* 라인 데이터가 12개가 넘는다면 행 추가
  PERFORM set_line_size.
* SO 라인 정보 세팅
  PERFORM fill_line_data.
*---------------------------------------------------------------------

* 기본 Sheet 1 세팅
  CALL METHOD OF gs_excel 'SHEETS' = gs_sheet EXPORTING #1 = 1.
  CALL METHOD OF gs_sheet 'SELECT' NO FLUSH.

* 특정 셀 좌측 정렬
  PERFORM set_cell_left USING 10 3.
  PERFORM set_cell_left USING 12 6.

* 생성 후 맨위로 스크롤 이동
  CALL METHOD OF gs_excel 'CELLS' = gs_cell EXPORTING #1 = 1 #2 = 2.

  CALL METHOD OF gs_cell 'SELECT'.

  SET PROPERTY OF gs_excel 'VISIBLE' = 0.

* 열 넓이 자동 맞춤 설정
  CALL METHOD OF gs_activesheet 'COLUMNS' = gs_cells.
  CALL METHOD OF gs_cells 'AUTOFIT' NO FLUSH.

* 열 넓이 최소 길이 설정 - 시작 칼럼, 끝 칼럼, 최소길이
  PERFORM set_column_min_length USING 'C' 'C' '20'.
  PERFORM set_column_min_length USING 'D' 'D' '40'.

* 시트 이름 변경
  SET PROPERTY OF gs_activesheet 'Name' = TEXT-n02.

* 저장
  CALL METHOD OF gs_excel 'ACTIVEWORKBOOK' = gs_workbook.
  CALL METHOD OF gs_workbook 'SAVE'.
  CALL FUNCTION 'CONTROL_FLUSH'.

* PDF 저장
  PERFORM convert_to_pdf.

* Free
  PERFORM free_object.

ENDFORM.

FORM download_template .

  DATA : ls_wwwdata_item TYPE wwwdatatab,
         lv_rc           TYPE sy-subrc.

* 파일 캐시 메모리 삭제
  CALL FUNCTION 'WS_FILE_DELETE'
    EXPORTING
      file   = gv_file
    IMPORTING
      return = lv_rc.

* 성공 아니면 메시지 처리
  IF ( lv_rc <> 0 ) AND
     ( lv_rc <> 1 ).
    MESSAGE s101 DISPLAY LIKE 'E'.
  ENDIF.

  SELECT SINGLE *
    INTO CORRESPONDING FIELDS OF ls_wwwdata_item
    FROM wwwdata
    WHERE objid = gv_form.

  CALL FUNCTION 'DOWNLOAD_WEB_OBJECT'
    EXPORTING
      key         = ls_wwwdata_item
      destination = gv_file.

ENDFORM.

                    gs_application 'ACTIVESHEET' = gs_activesheet.

FORM fill_company_data .

  DATA : lv_row TYPE i VALUE 5.
  CONSTANTS : lv_col TYPE i VALUE 5.
  FIELD-SYMBOLS : <fs_field> TYPE any.

* Excel 출력 기업 정보 3개
  DO 3 TIMES.

    ASSIGN COMPONENT sy-index OF STRUCTURE gs_company TO <fs_field>.

      EXIT.

    PERFORM fill_cells USING lv_row lv_col <fs_field>.
    UNASSIGN <fs_field>.

    lv_row += 1.

  ENDDO.

* Excel 출력 헤더 필드 8개 세팅
  DO 8 TIMES.

    ASSIGN COMPONENT sy-index OF STRUCTURE gs_header TO <fs_field>.
    IF <fs_field> IS NOT ASSIGNED.
      EXIT.
    ENDIF.

* 열, 행, 값 채우기
    PERFORM fill_cells USING lv_row lv_col <fs_field>.
    UNASSIGN <fs_field>.

* lv_col = 6이면 밑 줄 3번째로 이동
    IF lv_col = 6.
      lv_row += 1.
      lv_col = 3.

    ELSE.
      lv_col += 3.
    ENDIF.

  ENDDO.

*&---------------------------------------------------------------------*
*& Form fill_line_data
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM fill_line_data .

  DATA : lv_row  TYPE i VALUE 16,
         lv_col  TYPE i VALUE 2,
         lv_deli TYPE c.

  DATA : ls_excel(1024) TYPE c,
         lt_excel       LIKE TABLE OF ls_excel,
         lv_rc          TYPE i.

  FIELD-SYMBOLS : <fs_field> TYPE any.

* Excel 셀 구분자(#)
  lv_deli = cl_abap_char_utilities=>horizontal_tab.

* 라인 아이템 -> Char 형으로 변경 및 Char Table로 변경
  LOOP AT gt_line ASSIGNING FIELD-SYMBOL(<fs_line>).

* 앞에서부터 9개의 필드만 Excel로 출력
*-- 30 # AMC16VA2PSW10I-C1 # AMC16VA2PSW10: PAC-A,6.5KW,2+1,D/SILVER, # 11,100 # KRW # 11 # EA # 122,100 # 자재창고
    DO 9 TIMES.

      ASSIGN COMPONENT sy-index OF STRUCTURE <fs_line> TO <fs_field>.
      IF <fs_field> IS NOT ASSIGNED.
        EXIT.
      ENDIF.

* 각 필드 중간 Deli 필드를 추가하여 셀 구분
      IF sy-index = 1.
        ls_excel = <fs_field>.
      ELSE.
        ls_excel = |{ ls_excel } { lv_deli } { <fs_field> }|.
      ENDIF.

      UNASSIGN <fs_field>.

    ENDDO.

    APPEND ls_excel TO lt_excel.
    CLEAR ls_excel.

  ENDLOOP.

* lt_excel 데이터 클립보드에 전체 복사
  CALL METHOD cl_gui_frontend_services=>clipboard_export
    EXPORTING
      no_auth_check        = ' '
    IMPORTING
      data                 = lt_excel
    CHANGING
      rc                   = lv_rc
    EXCEPTIONS
      cntl_error           = 1                " Control error
      error_no_gui         = 2                " No GUI available
      not_supported_by_gui = 3                " GUI does not support this
      no_authority         = 4                " Authorization check failed
      OTHERS               = 5.

* 붙여넣을 셀 지정
  CALL METHOD OF gs_excel 'CELLS' = gs_cell
     EXPORTING
       #1 = lv_row
       #2 = lv_col.

* 셀 선택 지정
  CALL METHOD OF gs_cell 'SELECT'.

* 클립보드 내용 붙여넣기
  CALL METHOD OF gs_activesheet 'PASTE'.

ENDFORM.

FORM set_header_field .

* 바이어 데이터 세팅
  DATA(lv_kunnr) = |{ gs_header-kunnr ALPHA = OUT }|.
  CONDENSE lv_kunnr NO-GAPS.
  gs_header-buyer = |{ lv_kunnr }({ gs_header-name1 })|.

* 헤더 정보에 현재 날짜 세팅
  gs_header-dudat = sy-datum.

* Total(금액 + 통화키) 필드 세팅
  WRITE gs_header-netwr CURRENCY gs_header-waerk TO gs_header-total
                                                 NO-GAP.
  CONDENSE gs_header-total NO-GAPS.
  gs_header-total = |{ gs_header-total } { gs_header-waerk }|.

ENDFORM.
*&---------------------------------------------------------------------*
*& Form set_line_field
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM set_line_field .

  LOOP AT gt_line ASSIGNING FIELD-SYMBOL(<fs_line>).
* 금액 필드 통화키에 맞게 환산
    WRITE <fs_line>-netpr CURRENCY <fs_line>-waerk
                                   TO <fs_line>-price_unit.
    CONDENSE <fs_line>-price_unit NO-GAPS.

* 수량 필드 수량단위에 맞게 환산
    WRITE <fs_line>-kwmeng UNIT <fs_line>-meins
                             TO <fs_line>-quantity.
    CONDENSE <fs_line>-quantity NO-GAPS.

* 총 금액 계산
    WRITE <fs_line>-total CURRENCY <fs_line>-waerk
                                TO <fs_line>-price_total.
    CONDENSE <fs_line>-price_total NO-GAPS.

  ENDLOOP.

ENDFORM.

*& Form convert_to_pdf
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM convert_to_pdf .

  DATA : lv_filename_pdf TYPE string.

* 앞뒤 공백/탭 제거
  CLEAR lv_filename_pdf.
  lv_filename_pdf = condense( gv_file ).

* xlsx, xls 파일 확장자 pdf로 변경
  REPLACE REGEX '\.(xlsx|xls)$' IN lv_filename_pdf WITH '.pdf' IGNORING CASE.
  GET PROPERTY OF gs_excel 'WORKBOOKS' = gs_workbook
    EXPORTING #1 = 1.

*-- PDF 가로로 조정 및 Excel 파일 사이즈에 맞게 조절
  " 페이지 설정 객체 가져오기
  " 첫 번째 Worksheet 가져오기
  GET PROPERTY OF gs_workbook 'WORKSHEETS' = gs_worksheet EXPORTING #1 = 1.
  GET PROPERTY OF gs_worksheet 'PAGESETUP' = gs_pagesetup.
  SET PROPERTY OF gs_pagesetup 'ORIENTATION' = 2.                           " 2 = Landscape, 1 = Portrait
*  GET PROPERTY OF gs_excel 'WORKBOOKS' = gs_workbook EXPORTING #1 = 1.
*  SET PROPERTY OF gs_pagesetup 'PAPERSIZE' = 8.                             " 9->A4, 8->A3... 등등
*  SET PROPERTY OF gs_pagesetup 'Zoom' = abap_False.
*  SET PROPERTY OF gs_pagesetup 'FitToPagesWide' = 1.
*  SET PROPERTY OF gs_pagesetup 'FitToPagesTall' = 0.

* FitToPagesWidth, FitToPagesTall을 뒤에 {1,0}으로 세팅할 수 있음 다른 파라미터는 몰루
  CALL METHOD OF gs_excel 'EXECUTEEXCEL4MACRO'
    EXPORTING
      #1 = 'PAGE.SETUP(,,,,,,,,,,,,{1,0})'.

* PDF 추출
  CALL METHOD OF gs_workbook 'EXPORTASFIXEDFORMAT'
    EXPORTING
      #1 = '0'
      #2 = lv_filename_pdf.

* 연결 해제
  CALL METHOD OF gs_workbook 'CLOSE'
    EXPORTING
      #1 = 0.

* 엑셀 종료
  CALL METHOD OF gs_excel 'QUIT'.

ENDFORM.

*&---------------------------------------------------------------------*
*& Form set_column_min_length
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM set_column_min_length USING pv_start pv_end pv_length.

  DATA: lo_range TYPE ole2_object,
        lo_cols  TYPE ole2_object,
        lo_col   TYPE ole2_object,
        lv_cnt   TYPE i,
        lv_width TYPE f.

  DATA : lv_range TYPE string.

  lv_range = |${ pv_start }:${ pv_end }|.

  " 지정 범위 칼럼 대상
  CALL METHOD OF gs_activesheet 'RANGE' = lo_range
    EXPORTING
      #1 = lv_range.
  " 전체 열 한번에 컨트롤
  GET PROPERTY OF lo_range 'ENTIRECOLUMN' = lo_cols.

  " 1) 지정 범위 컬럼 추출
  GET PROPERTY OF lo_cols 'COUNT' = lv_cnt.

  " 2) 컬럼들 최소 길이 보장
  DO lv_cnt TIMES.
    CALL METHOD OF lo_cols 'ITEM' = lo_col EXPORTING #1 = sy-index.
    GET PROPERTY OF lo_col 'COLUMNWIDTH' = lv_width.
    IF lv_width LT pv_length.
      SET PROPERTY OF lo_col 'COLUMNWIDTH' = pv_length.
    ENDIF.
  ENDDO.

  CALL FUNCTION 'CONTROL_FLUSH'.

ENDFORM.
