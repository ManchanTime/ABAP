FORM set_screen_data .

  LOOP AT SCREEN.

    IF screen-name = 'PA_CARR'.
      screen-input = 0.
    ELSEIF screen-name = 'PA_BUKRS'.
      screen-input = 0.
    ELSEIF screen-name = 'PA_GJAHR'.
      screen-input = 0.
    ELSEIF screen-name = 'PA_BELNR'.
      IF pa_edit = 'X'.
        screen-input = 1.
      ELSE.
        screen-input = 0.
      ENDIF.
    ENDIF.

    MODIFY SCREEN.
  ENDLOOP.

ENDFORM.
