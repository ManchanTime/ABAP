FORM get_tax_data .

  DATA : lv_cnt TYPE i.

*-- Get tax code data
  CLEAR gt_t007s.
  SELECT mwskz text1
    INTO CORRESPONDING FIELDS OF TABLE gt_t007s
    FROM t007s
    WHERE kalsm = pa_kalsm
      AND mwskz IN so_mwskz.

  IF ( gt_t007s IS INITIAL ).
    MESSAGE s009 DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

  lv_cnt = lines( gt_t007s ).

  MESSAGE s008 WITH lv_cnt.
