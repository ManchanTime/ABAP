FORM get_main_data .

  READ TABLE s_ersda ASSIGNING FIELD-SYMBOL(<fs_ersda>).
  IF ( <fs_ersda> IS ASSIGNED ) AND
     ( <fs_ersda>-low IS NOT INITIAL ) AND
     ( <fs_ersda>-high IS INITIAL ).

    <fs_ersda>-option = 'BT'.
    <fs_ersda>-high = sy-datlo.
    READ TABLE s_ersda INDEX 1.

  ENDIF.

  CLEAR gt_data.
  WITH +mkal AS (
    SELECT matnr, werks, MIN( verid ) AS verid
      FROM mkal
     WHERE stlal = @s_stlal-low
       AND stlan = @s_stlan-low
       AND plnty = @s_plnty-low
     GROUP BY matnr, werks
  )
  SELECT b~werks, a~matnr, a~mtart, a~ersda, a~matkl, a~meins, a~pstat,
         b~dispo, c~bklas, b~beskz,
         CASE WHEN a~pstat     LIKE '%A%' THEN '@5B@'
              WHEN e~pstat NOT LIKE '%A%' THEN '@00@' END AS wkpln,
         CASE WHEN a~pstat     LIKE '%B%' THEN '@5B@'
              WHEN e~pstat NOT LIKE '%B%' THEN '@00@' END AS acctg,
*         CASE WHEN a~pstat     LIKE '%C%' THEN '@5B@'
*              WHEN e~pstat NOT LIKE '%C%' THEN '@00@' END AS categ,
         CASE WHEN a~pstat     LIKE '%D%' THEN '@5B@'
              WHEN e~pstat NOT LIKE '%D%' THEN '@00@' END AS mrppr,
         CASE WHEN a~pstat     LIKE '%E%' THEN '@5B@'
              WHEN e~pstat NOT LIKE '%E%' THEN '@00@' END AS purch,
         CASE WHEN a~pstat     LIKE '%F%' THEN '@5B@'
              WHEN e~pstat NOT LIKE '%F%' THEN '@00@' END AS prdst,
         CASE WHEN a~pstat     LIKE '%G%' THEN '@5B@'
              WHEN e~pstat NOT LIKE '%G%' THEN '@00@' END AS costg,
         CASE WHEN a~pstat     LIKE '%K%' THEN '@5B@'
              WHEN e~pstat NOT LIKE '%K%' THEN '@00@' END AS basdt,
         CASE WHEN a~pstat     LIKE '%L%' THEN '@5B@'
              WHEN e~pstat NOT LIKE '%L%' THEN '@00@' END AS storg,
         CASE WHEN a~pstat     LIKE '%P%' THEN '@5B@'
              WHEN e~pstat NOT LIKE '%P%' THEN '@00@' END AS forct,
         CASE WHEN a~pstat     LIKE '%Q%' THEN '@5B@'
              WHEN e~pstat NOT LIKE '%Q%' THEN '@00@' END AS qltyc,
         CASE WHEN a~pstat     LIKE '%S%' THEN '@5B@'
              WHEN e~pstat NOT LIKE '%S%' THEN '@00@' END AS whmgt,
         CASE WHEN a~pstat     LIKE '%V%' THEN '@5B@'
              WHEN e~pstat NOT LIKE '%V%' THEN '@00@' END AS sales,
         CASE WHEN a~pstat     LIKE '%X%' THEN '@5B@'
              WHEN e~pstat NOT LIKE '%X%' THEN '@00@' END AS plinv,
         CASE WHEN a~pstat     LIKE '%Z%' THEN '@5B@'
              WHEN e~pstat NOT LIKE '%Z%' THEN '@00@' END AS stloc,
         CASE WHEN b~matnr EQ     g~matnr THEN '@5B@' END AS bilom,
         CASE WHEN b~matnr EQ     f~matnr THEN '@5B@' END AS routg,
         CASE WHEN b~matnr EQ     h~matnr THEN '@5B@' END AS prodv
    FROM mara AS a INNER JOIN marc AS b
                           ON a~matnr = b~matnr
                   INNER JOIN t001w AS d
                           ON b~werks = d~werks
                          AND d~spras = @sy-langu
                   INNER JOIN mbew AS c
                           ON a~matnr = c~matnr
                          AND d~bwkey = c~bwkey
                   INNER JOIN t134 AS e
                           ON a~mtart = e~mtart
                    LEFT JOIN mast AS g
                           ON b~matnr = g~matnr
                          AND b~werks = g~werks
                          AND g~stlan = @s_stlan-low
                          AND g~stlal = @s_stlal-low
                    LEFT JOIN mapl AS f
                           ON b~matnr = f~matnr
                          AND b~werks = f~werks
                          AND f~plnty = @s_plnty-low
                          AND f~plnal = '1'
                    LEFT JOIN +mkal AS h
                           ON b~matnr = h~matnr
                          AND b~werks = h~werks
    WHERE b~werks IN @s_werks
      AND a~mtart IN @s_mtart
      AND ersda   IN @s_ersda
      AND a~matnr IN @s_matnr
      AND dismm   IN @s_dismm
      AND b~beskz IN @s_beskz
      AND c~bklas IN @s_bklas
    ORDER BY b~matnr, b~werks
    INTO CORRESPONDING FIELDS OF TABLE @gt_data.

  IF sy-subrc = 0.
    MESSAGE s000 WITH sy-dbcnt.
  ELSE.
    MESSAGE s101 DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

ENDFORM.

FORM set_fcat USING pv_key pv_field pv_table pv_just pv_emph
                    pt_fcat TYPE lvc_t_fcat.

  DATA : ls_fcat TYPE lvc_s_fcat.

  ls_fcat = VALUE #( key       = pv_key
                     fieldname = pv_field
                     ref_table = pv_table
                     just      = pv_just
                     emphasize = pv_emph ).

  ls_fcat = VALUE #( BASE ls_fcat
                      coltext = SWITCH #( pv_field WHEN 'WKPLN' THEN '작업일정 계획'
                                                   WHEN 'ACCTG' THEN '회계'
                                                   WHEN 'MRPPR' THEN 'MRP'
                                                   WHEN 'PURCH' THEN '구매'
                                                   WHEN 'PRDST' THEN '생산 자원/툴'
                                                   WHEN 'COSTG' THEN '원가계산'
                                                   WHEN 'BASDT' THEN '기본 데이터'
                                                   WHEN 'STORG' THEN '저장'
                                                   WHEN 'FORCT' THEN '예측'
                                                   WHEN 'QLTYC' THEN '품질 관리'
                                                   WHEN 'SALES' THEN '매출액(영업조직)'
                                                   WHEN 'WHMGT' THEN '창고관리'
                                                   WHEN 'PLINV' THEN '플랜트 재고'
                                                   WHEN 'STLOC' THEN '저장위치 재고'
                                                   WHEN 'BILOM' THEN 'BOM'
                                                   WHEN 'ROUTG' THEN 'ROUTING'
                                                   WHEN 'PRODV' THEN '생산 버전' )
                      style   = SWITCH #( pv_field WHEN 'MATNR' THEN '00000200' ) ).

  APPEND ls_fcat TO gt_fcat.

ENDFORM.

FORM create_object .

  CREATE OBJECT go_docking
    EXPORTING
      side                        = go_docking->dock_at_left
      extension                   = 5000
    EXCEPTIONS
      cntl_error                  = 1
      cntl_system_error           = 2
      create_error                = 3
      lifetime_error              = 4
      lifetime_dynpro_dynpro_link = 5
      OTHERS                      = 6.

  CREATE OBJECT go_split
    EXPORTING
      parent            = go_docking
      rows              = 2
      columns           = 1
    EXCEPTIONS
      cntl_error        = 1
      cntl_system_error = 2
      OTHERS            = 3.

  CALL METHOD go_split->set_row_height
    EXPORTING
      id                = 1
      height            = 30
    EXCEPTIONS
      cntl_error        = 1
      cntl_system_error = 2
      OTHERS            = 3.

  CALL METHOD go_split->get_container
    EXPORTING
      row       = 1
      column    = 1
    RECEIVING
      container = go_cont_page.

  CALL METHOD go_split->get_container
    EXPORTING
      row       = 2
      column    = 1
    RECEIVING
      container = go_cont_main.


  CREATE OBJECT go_grid_main
    EXPORTING
      i_parent          = go_cont_main
    EXCEPTIONS
      error_cntl_create = 1
      error_cntl_init   = 2
      error_cntl_link   = 3
      error_dp_create   = 4
      OTHERS            = 5.

  CREATE OBJECT go_grid_page
    EXPORTING
      style = 'ALV_GRID'.

ENDFORM.

FORM handle_top_of_page USING po_dyndoc_id TYPE REF TO cl_dd_document.

  DATA : lo_table   TYPE REF TO cl_dd_table_element,
         lo_column  TYPE REF TO cl_dd_area,
         lo_column2 TYPE REF TO cl_dd_area.

  DATA : lv_value TYPE string.

  po_dyndoc_id->add_table(
    EXPORTING
      no_of_columns = 2
      border        = '1'
      width         = '30%'
    IMPORTING
      table = lo_table
  ).

  " 테이블 세트
  lo_table->add_column( EXPORTING width = '50%' IMPORTING column = lo_column ).
  lo_table->add_column( EXPORTING width = '50%' IMPORTING column = lo_column2 ).

  PERFORM make_value USING lv_value s_werks-low s_werks-high.
  lo_column->add_text( EXPORTING text = '플랜트' ).
  lo_column2->add_text( EXPORTING text = |{ lv_value }| ).

  lo_table->new_row( ).
  PERFORM make_value USING lv_value s_mtart-low s_mtart-high.
  lo_column->add_text( EXPORTING text = '자재유형' ).
  lo_column2->add_text( EXPORTING text = |{ lv_value }| ).

  lo_table->new_row( ).
  PERFORM make_value_date USING lv_value s_ersda-low s_ersda-high.
  lo_column->add_text( EXPORTING text = '생성일' ).
  lo_column2->add_text( EXPORTING text = |{ lv_value }| ).

  lo_table->new_row( ).
  PERFORM make_value USING lv_value s_matnr-low s_matnr-high.
  lo_column->add_text( EXPORTING text = '자재' ).
  lo_column2->add_text( EXPORTING text = |{ lv_value }| ).

  lo_table->new_row( ).
  PERFORM make_value USING lv_value s_dismm-low s_dismm-high.
  lo_column->add_text( EXPORTING text = 'MRP 관리자' ).
  lo_column2->add_text( EXPORTING text = |{ lv_value }| ).

  lo_table->new_row( ).
  PERFORM make_value USING lv_value s_beskz-low s_beskz-high.
  lo_column->add_text( EXPORTING text = '조달유형' ).
  lo_column2->add_text( EXPORTING text = |{ lv_value }| ).

  lo_table->new_row( ).
  PERFORM make_value USING lv_value s_bklas-low s_bklas-high.
  lo_column->add_text( EXPORTING text = '평가 클래스' ).
  lo_column2->add_text( EXPORTING text = |{ lv_value }| ).


  po_dyndoc_id->merge_document( ).
  po_dyndoc_id->display_document( EXPORTING parent = go_cont_page ).

ENDFORM.
*&---------------------------------------------------------------------*
*& Form make_Value
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*&      --> S_WERKS_LOW
*&      --> S_WERKS_HIGH
*&---------------------------------------------------------------------*
FORM make_value USING pv_value pv_low pv_high.

  CLEAR pv_value.
  IF pv_low IS INITIAL.
    pv_value = '*'.
  ELSEIF pv_high IS NOT INITIAL.
    pv_value = |{ pv_low } ~ { pv_high }|.
  ELSE.
    pv_value = |{ pv_low }|.
  ENDIF.

ENDFORM.
*&---------------------------------------------------------------------*
*& Form make_value_date
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*&      --> LV_VALUE
*&      --> S_ERSDA_LOW
*&      --> S_ERSDA_HIGH
*&---------------------------------------------------------------------*
FORM make_value_date USING pv_value pv_low pv_high.

  CLEAR pv_value.
  IF pv_low IS INITIAL.
    pv_value = '*'.
  ELSEIF pv_high IS NOT INITIAL.
    pv_value = |{ pv_low(4) }.{ pv_low+4(2) }.{ pv_low+6(2) } ~ { pv_high(4) }.{ pv_high+4(2) }.{ pv_high+6(2) }|.
  ELSE.
    pv_value = |{ pv_low(4) }.{ pv_low+4(2) }.{ pv_low+6(2) }|.
  ENDIF.

ENDFORM.

*& Form handle_double_click
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*&      --> E_COLUMN
*&      --> E_ROW
*&---------------------------------------------------------------------*
FORM handle_double_click USING pv_column pv_row.

  READ TABLE gt_data ASSIGNING FIELD-SYMBOL(<fs_data>) INDEX pv_row.
  IF sy-subrc = 0.

    CASE pv_column.
      WHEN 'MATNR'.
        PERFORM set_parameter USING <fs_data>-matnr 'MAT'.
        PERFORM set_parameter USING 'K'             'MXX'.
        CALL TRANSACTION 'MM03' AND SKIP FIRST SCREEN.
      WHEN OTHERS.
        ASSIGN COMPONENT pv_column OF STRUCTURE <fs_data> TO FIELD-SYMBOL(<fs_value>).
        IF ( <fs_value> IS ASSIGNED ) AND
           ( <fs_value> = icon_led_green ).

          PERFORM set_parameter USING <fs_data>-matnr 'MAT'.
          CASE pv_column.
            WHEN 'WKPLN'.
              PERFORM set_parameter USING <fs_data>-werks 'WRK'.
              PERFORM set_parameter USING 'A'             'MXX'.
            WHEN 'ACCTG'.
              PERFORM set_parameter USING <fs_data>-werks 'WRK'.
              PERFORM set_parameter USING 'B'             'MXX'.
            WHEN 'MRPPR'.
              PERFORM set_parameter USING 'D'             'MXX'.
            WHEN 'PURCH'.
              PERFORM set_parameter USING 'E'             'MXX'.
            WHEN 'PRDST'.
              PERFORM set_parameter USING 'F'             'MXX'.
            WHEN 'COSTG'.
              PERFORM set_parameter USING 'G'             'MXX'.
            WHEN 'BASDT'.
              PERFORM set_parameter USING 'K'             'MXX'.
            WHEN 'STORG'.
              PERFORM set_parameter USING 'L'             'MXX'.
            WHEN 'FORCT'.
              PERFORM set_parameter USING 'P'             'MXX'.
            WHEN 'QLTYC'.
              PERFORM set_parameter USING 'Q'             'MXX'.
            WHEN 'SALES'.
              PERFORM set_parameter USING 'V'             'MXX'.
            WHEN 'WHMGT'.
              PERFORM set_parameter USING 'S'             'MXX'.
            WHEN 'PLINV'.
              PERFORM set_parameter USING 'D'             'MXX'.
            WHEN 'STLOC'.
              PERFORM set_parameter USING 'Z'             'MXX'.
            WHEN 'BILOM'.
*              PERFORM set_parameter USING 'A'             'MXX'.
            WHEN 'ROUTG'.
*              PERFORM set_parameter USING 'A'             'MXX'.
            WHEN 'PRODV'.
*              PERFORM set_parameter USING 'AD'             'MXX'.
          ENDCASE.

          CALL TRANSACTION 'MM03' AND SKIP FIRST SCREEN.
        ENDIF.

    ENDCASE.

  ENDIF.

ENDFORM.
