FORM set_init_data .

  pa_bukrs = '0001'.
  pa_gjahr = sy-datum(4).

ENDFORM.
*&---------------------------------------------------------------------*
*& Form get_function_data
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM get_function_data .

  CALL FUNCTION 'ZFM_CL1_16_08'
    EXPORTING
      iv_bukrs   = pa_bukrs
      iv_gjahr   = pa_gjahr
    IMPORTING
      es_header  = gs_header
      ev_message = gv_message
    TABLES
      et_item    = gt_item.


ENDFORM.
*&---------------------------------------------------------------------*
*& Form diplay_data
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM diplay_data.

  DATA : lv_dbcnt TYPE sy-dbcnt.

  lv_dbcnt = LINES( gt_item ).
  IF lv_dbcnt <> 0.
    MESSAGE s008 WITH lv_dbcnt.
  ELSE.
    MESSAGE s000 WITH gv_message DISPLAY LIKE 'E'.
  ENDIF.

  cl_demo_output=>write( gs_header ).
  cl_demo_output=>display( gt_item ).

ENDFORM.
